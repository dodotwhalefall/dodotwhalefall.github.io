### 1. 非递归中序遍历
```cpp
void Inorder(Bitree T){
   InitStack(s);        //初始化栈以及所需变量
   Bitree p=T;
   while(p||!IsEmpty(s)){//循环条件
       if(p){            //执行单元的条件
           Push(S,p);    //左全入栈
           p=p->lchild
       }
       else{
           Pop(S,p);visit(p);//左先访问然后根
           p=p->rchild;        //左子树和根节点的右子树都要访问写在一起
       }
   }
}
```
### 2. 非递归后序遍历
```Cpp
    void Postorder(Bitree T){
        InitStack(S);
        BiNode *p=T;   //工作指针
        r=NULL;         //区分前一个访问的是左还是右
        while(p||!IsEmpty(S))//循环条件
        {
            if(p){             //老三样 先左全入
                Push(S,p);
                p=p->lchild;
            }
            else{
                GetTop(S,p);//取栈顶
                if(p->rchild&&p->rchild!=r){//在访问根节点起作用，若是右则不执行
                    p=p->rchild;
                    Push(S,p);//右节点在根前面插队
                    p=p->lchild;
                }
                else{//最后
                    pop(S,p);
                    visit(p->data);//左右节点都是确定为叶子才访问，根节点是确定左右都访问过才访问
                    r=p;
                    p=NULL;
                }
            }
        }
    }
```

void LevelOrder(Bitree T){
    InitQueue(Q);

### 6.二分查找
#### 6.1 普通细节但是不稳定
```cpp
int binarySearch(int[] nums, int target) {
    int left = 0; 
    int right = nums.length - 1; // 相当于闭区间

    while(left <= right) {//相当于中止条件是左大于右
        int mid = left + (right - left) / 2;//避免溢出
        if(nums[mid] == target)
            return mid; 
        else if (nums[mid] < target)
            left = mid + 1; // 这是合理的搜索区间
        else if (nums[mid] > target)
            right = mid - 1; // 注意
    }
    return -1;
}
```
#### 6.2 寻找最左侧边界
```cpp
int left_bound(int[] nums, int target) {
    int left = 0, right = nums.length - 1;
    // 搜索区间为 [left, right]
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (nums[mid] < target) {
            // 搜索区间变为 [mid+1, right]
            left = mid + 1;
        } else if (nums[mid] > target) {
            // 搜索区间变为 [left, mid-1]
            right = mid - 1;
        } else if (nums[mid] == target) {
            // 收缩右侧边界
            right = mid - 1;
        }
    }
    // 检查出界情况
    if (left >= nums.length || nums[left] != target)
        return -1;
    return left;
}
```
